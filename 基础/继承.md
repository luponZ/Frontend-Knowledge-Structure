> javaScript本质上是没有真正的继承的(继承意味着一个类的成员变量和成员方法复制到另一个类中),我们这里所说的继承其实利用原型链将两个对象连接起来,A通过委托访问其"继承对象B"的属性和方法
> 这种链式继承当访问的属性过于底层的话, 性能会是问题


### 原型链继承

```js
    function Father(name) {
        this.name = name;
    }

    Father.prototype.say = function () {
        console.log(this.name)
    }

    function Son(age) {
        this.age = age;
    }

    // 把父类的私有+公有的属性和方法，都作为子类公有的属性
    Son.prototype = new Father();
    // 重置构造属性
    // ES5+
    Object.defineProperty(Son, constructor, {
        enumerable: false,
        value: Son
    })
    // ES3-
    Son.prototype.constructor = Son;

    new Son();
```
缺点: 
1. 父类的引用属性会被所有子类实例共享
2. 子类构建实例时不能向父类传递参数

### 构造函数继承
```js
    function Father(name) {
        this.name = name;
    }

    Father.prototype.say = function () {
        console.log(this.name)
    }

    function Son(age, name) {
        this.age = age;
        // 调用父级,改变this
        Father.call(this, name)
    }

    new Son();
```
缺点:
1. 父类的公有属性和公有方法无法继承

### 组合继承(伪经典继承)
```js
    function Father(name) {
        this.name = name;
    }

    Father.prototype.say = function () {
        console.log(this.name)
    }

    function Son(age, name) {
        this.age = age;
        // 调用父级,改变this
        Father.call(this, name)
    }

    Son.prototype = new Father();

    new Son();
```
缺点: 
1. 多次继承父级私有属性,造成性能浪费

### 寄生式继承
```js
    // Object.create原理
    function objectCreate(ob) {
        function F() { };
        F.prototype = ob;
        return new F();
    }

    function cloneNode(ob) {
        const clone = objectCreate(ob);
        clone.say = function () { console.log(this.hab) };
        return clone;
    }

    function Father(name) {
        this.name = name;
        this.hab = [];
    }

    Father.prototype.add = function (i) {
        this.hab.push(i)
    }

    const son = cloneNode(new Father("hello"));
    son.add("11");
    son.say();
```

### 寄生组合继承(经典继承)
```js
    function inheritPrototype(subType, superType) {
        const proto = Object.create(superType.prototype);
        Object.defineProperty(proto, constructor, {
            enumerable: false,
            value: subType
        })
        subType.prototype = proto;
    }

    function Father(name) {
        this.name = name;
    }

    Father.prototype.say = function () {
        console.log(this.name)
    }

    function Son(age, name) {
        this.age = age;
        Father.call(name);
    }

    inheritPrototype(Son, Father);
    Son.prototype.say = function () {};

```
