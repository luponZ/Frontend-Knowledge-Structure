function curry (fn, currArg) {
    return function () {
        let args = Array.prototype.slice(arguments);
        if (currArg) {
            args = [currArg].concat(args);
        }
        if (args.length < fn.length) {
            return curry(fn, args);
        }
        return fn.apply(null, args)
    }
curry((a,b,c) => a + b + c)
