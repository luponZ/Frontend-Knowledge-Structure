### 防抖
> 在事件被触发n秒后再执行回调，如果在这n秒内又被触发，则取消之前的事件回调  重新计时。
```js
    function debounce(fn, time, immediate) {
        let timer = null;
        return function() {
            const self = this;
            const args = arguments;
            if (immediate && !timer) {
                fn.apply(self, args)
            }
            if (timer) {
                clearTimeout(timer)
            }
            timer = setTimeout(() => {
                fn.apply(self, args)
            }, time)
        }
    }
```
### 节流
> 规定一个单位时间，在这个单位时间内，只能有一次触发事件的回调函数执行，如果在同一个单位时间内某事件被触发多次，只有一次能生效
```js
    function throttle(fn, time, wait = 0) {
        let previous = null;
        let timer = null;
        return function () {
            const self = this;
            const args = arguments;
            const now = Date.now();
            previous = previous ? previous : now;
            if (now - previous >= time) {
                clearTimeout(timer);
                fn.apply(self, args);
                previous = Date.now();
            } else {
                clearTimeout(timer);
                setTimeout(() => {
                    fn.apply(self, args)
                }, wait)
            }
        }
    }
```
