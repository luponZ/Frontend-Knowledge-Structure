##### 1. 冒泡排序
```js
// 普通版本
function bubble(arr) {
    let len = arr.length;
    for(let i = 0; i < len; i++) {
        for(let j = 0; j < len - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                let mi = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = mi;
            }
        }
    }
    return arr;
}
// 改进版本
function bubbleByIndex(arr) {
    let i = arr.length - 1;
    while(i > 0) {
        let pos = 0;
        for(let j = 0; j < i; j++) {
            if (arr[j] > arr[j + 1]) {
                let mi = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = mi;
                pos = j;
            }
        }
        i = pos
    }
    return arr;
}

// bubble: 0.643ms
// bubbleByIndex: 0.009ms
```
##### 2. 快速排序
```js
function fastSort(arr) {
    if (arr.length <= 1) {
        return arr
    } else {
        let midIndex = Math.floor(arr.length / 2);
        let mid = arr.splice(midIndex, 1)[0];
        let left = [];
        let right = [];
        let len = arr.length;
        for (let i = 0; i < len; i++) {
            if (arr[i] <= mid) {
                left.push(arr[i])
            } else {
                right.push(arr[i])
            }
        }
        return fastSort(left).concat([mid], fastSort(right));
    }
}
```

#### 2. 选择排序
##### 1. 简单选择排序
```js
function selectionSort(arr) {
    let len = arr.length;
    let minIndex = null;
    let mid = null;
    for(let i = 0; i < len - 1; i++) {
        minIndex = i;
        for(let j = i + 1; j < len; j++) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j;
            }
        }
        mid = arr[i];
        arr[i] = arr[minIndex];
        arr[minIndex] = mid;
    }
    return arr
}
```
